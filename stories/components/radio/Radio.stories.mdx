## Label

Targets the immediate `<span>`, `<label>` or `.radio-label` element after the `input.radio` to apply the label styling.

<div className="pt-4 inline-flex items-center gap-4">
  <Radio id="superman" name="hero" label="Superman" />
  <Radio id="batman" name="hero" label="Batman" />
</div>

<Source
  format
  language="html"
  code={detent`
    <label for="superman" class="inline-flex items-center">
      <input id="superman" name="hero" type="radio" class="radio" />
      <span>Superman</span>
    </label>

    <div class="inline-flex items-center">
      <input id="batman" name="hero" type="radio" class="radio" />
      <label for="batman">Batman</label>
    </div>

`}
/>
