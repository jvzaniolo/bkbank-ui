## Label

Targets the immediate `<label>`, `<span>` or `.switch-label` element after the `input.switch` to apply the label styling.

<div className="pt-4">
  <Switch id="label" label="Switch" />
</div>

<Source
  format
  language="html"
  code={detent`
    <label class="inline-flex">
      <input type="checkbox" className="switch" />
      <span>Switch</span>
    </label>
  `}
/>
