# Checkbox

## Default

<div className="pt-4">
  <input type="checkbox" class="checkbox" />
</div>

<Source
  format
  language="html"
  code={detent`<input type="checkbox" class="checkbox" />`}
/>

## Label

Targets the immediate `<span>`, `<label>` or `.checkbox-label` element after the `input.checkbox` to apply the label styling.

<div className="pt-4">
  <Checkbox id="label" />
</div>

<Source
  format
  language="html"
  code={detent`
    <label for="checkbox" class="inline-flex items-center">
      <input id="checkbox" type="checkbox" class="checkbox" />
      <span>Checkbox</span>
    </label>

    <!-- or -->

    <div class="inline-flex items-center">
      <input id="checkbox" type="checkbox" class="checkbox" />
      <label for="checkbox">Checkbox</label>
    </div>

    <!-- or -->

    <label for="checkbox" class="inline-flex items-center">
      <input id="checkbox" type="checkbox" class="checkbox" />
      <p class="checkbox-label">Checkbox</p>
    </label>

`}
/>
